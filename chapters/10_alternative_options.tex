\chapter{Alternative Options for M.Tech}
\label{chap:alternatives}

\begin{protip}[Second Chances Exist]
  These options provide a viable second chance but can be equally challenging
  to secure in their own right. Explore them seriously if \GATE scores fall
  short of top IIT/NIT cutoffs.
\end{protip}

\section{IIIT Hyderabad --- PGEE}
\label{sec:iiith}

IIIT Hyderabad conducts its own \textbf{PG Entrance Examination (PGEE)} for:

\begin{pfChecklist}
  \item M.Tech
  \item Master of Science by Research
  \item Ph.D.
\end{pfChecklist}

\begin{pfActionList}

\section{Other Alternatives}
\label{sec:other-alternatives}

\begin{tblr}{
    colspec = {X[3,l] X[5,l]},
    hlines, vlines,
    row{1} = {bg=pfBlue, fg=white, font=\sffamily\bfseries},
    row{2,4,6,8,10} = {bg=pfGrey},
  }
  \toprule
  Institution & Notes \\
  \midrule
  \inst{Indian Statistical Institute (ISI)}
    & Admission via GATE (high cutoff) or separate written test + interview. \\
  \inst{VIT Vellore (VITMEE)}
    & Own entrance exam (VITMEE 2025 date: 20 April 2025). Good option for
      students who miss the GATE cutoff. \\
  \inst{DA-IICT Gandhinagar}
    & Separate entrance exam; no IIT tag but good placements; strong CS and ECE
      programs. \\
  \inst{NUS / Germany (MS abroad)}
    & Excellent option for high achievers considering global education; no GATE
      needed; requires GRE/IELTS in some cases. \\
  \inst{Jadavpur University / DTU}
    & Comparable to mid-tier NITs; own entrance process. \\
  \bottomrule
\end{tblr}

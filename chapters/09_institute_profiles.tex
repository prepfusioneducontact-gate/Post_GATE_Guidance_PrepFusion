\chapter{Institute-Wise Program Profiles}
\label{chap:institutes}

\begin{protip}[How to Use This Chapter]
  Each section below lists the M.Tech/MS programs available, the mode of
  selection (GATE-only vs.\ Interview), and useful preparation links.
  Always verify on the official institute website as programs and selection
  criteria change each year.
\end{protip}

%% ─────────────────────────────────────────────────────────
\section{IISc Bangalore}
\label{sec:iisc}

\subsection*{M.Tech Programs (with Interview)}

All M.Tech and M.Tech Research programs at IISc require an interview.

\begin{pfChecklist}
  \item Microelectronics \& VLSI (MVLSI)
  \item Electronic Systems Engineering (ESE)
  \item Computer Science \& Engineering (CSE)
  \item Artificial Intelligence (AI)
  \item Signal Processing
  \item Computational Data Science (CDS)
\end{pfChecklist}

\begin{eligalert}[IISc Cutoffs]
  \begin{pfChecklist}
    \item MVLSI: \cutoff{900} \GATE score (UR)
    \item ESE: \cutoff{850} \GATE score (UR)
  \end{pfChecklist}
\end{eligalert}

  Historically, all General category seats in IIT Kanpur's VLSI M.Tech have been

\section{COAP Admission Flowchart}
\label{sec:coap-flowchart}

\begin{figure}[!ht]
  \centering
  \begin{tikzpicture}[
      node distance = 1.5cm and 0.5cm,
      every node/.style = {align=center},
    ]
    % Nodes
    \node[pfStep] (gate)    {\GATE Result \\ Declared};
    \node[pfStep, below=of gate]   (apply)  {Apply on IIT/IISc \\ Websites};
    \node[pfStep, below=of apply]  (coap)   {Register on \\ \COAP Portal};
    \node[pfDecision, below=of coap] (inter) {Interview \\ Required?};
    \node[pfStep, right=3cm of inter] (attend) {Attend \\ Interview};
    \node[pfStep, below=of inter]  (offer)  {Receive COAP \\ Offer};
    \node[pfDecision, below=of offer] (satisfied) {Satisfied \\ with Offer?};
    \node[pfStep, right=3cm of satisfied] (hold) {Hold (max \\ 2 rounds)};
    \node[pfStep, below=of satisfied]  (accept) {Accept \& \\ Freeze Offer};
    \node[pfStep, below=of accept]  (admit)  {Follow Institute \\ Admission Instructions};

    % Arrows
    \draw[pfArrow] (gate)      -- (apply);
    \draw[pfArrow] (apply)     -- (coap);
    \draw[pfArrow] (coap)      -- (inter);
    \draw[pfArrow] (inter)     -- node[above, font=\small\sffamily]{Yes} (attend);
    \draw[pfArrow] (inter)     -- node[left,  font=\small\sffamily]{No}  (offer);
    \draw[pfArrow] (attend)    |- (offer);
    \draw[pfArrow] (offer)     -- (satisfied);
    \draw[pfArrow] (satisfied) -- node[above, font=\small\sffamily]{No}  (hold);
    \draw[pfArrow] (satisfied) -- node[left,  font=\small\sffamily]{Yes} (accept);
    \draw[pfArrow] (hold)      |- (offer);
    \draw[pfArrow] (accept)    -- (admit);
  \end{tikzpicture}
  \caption{COAP Admission Process Flowchart}
  \label{fig:coap-flow}
\end{figure}
